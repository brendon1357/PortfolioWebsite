StockChecker.py 
from requests_html import HTMLSession
from bs4 import BeautifulSoup
import time
import re
import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
import ctypes

EMAIL = ""
PASSWORD = ""


def parse_availability(soup, card, given_price, count):
    for elem in soup.findAll('div', {'class': 'item-container'}):
        item = elem.find('p', class_='item-promo')
        if item is None or item.text != "OUT OF STOCK":
            price_container = elem.find('li', {'class': 'price-current'})
            price = price_container.text.strip('$').strip('–').replace(",", "")
            f_price = re.sub(r'\([^)]*\)', '', price)

            if float(f_price) <= given_price:
                count += 1
                print("IN STOCK " + card + ": " + str(count))
                return True

            else:
                count += 1
                print("IN STOCK " + card + " (TOO EXPENSIVE): " + str(count))

        else:
            count += 1
            print("SOLD OUT " + card + ": " + str(count))


def notify(card, url):
    s = smtplib.SMTP(host="smtp-mail.outlook.com", port=587)

    s.starttls()
    s.login(EMAIL, PASSWORD)

    msg = MIMEMultipart()

    msg["From"] = EMAIL
    msg["To"] = EMAIL
    msg["Subject"] = "Notification about RTX 3000 series card"

    msg.attach(MIMEText("Possible " + card + " in stock on newegg at " +
                        url))

    s.send_message(msg)
    s.quit()

    ctypes.windll.user32.MessageBoxW(0, "Possible " + card + " stock on newegg", "newegg notification", 1)


count = 0

end_loop = False

while not end_loop:
    session = HTMLSession()

    response_3070 = session.get("https://www.newegg.ca/p/pl?d=rtx+3070+graphics+cards")

    soup_3070 = BeautifulSoup(response_3070.content, 'html.parser')

    if parse_availability(soup_3070, "3070", 1500.00, count):
        notify("3070", "https://www.newegg.ca/p/pl?d=RTX+3070+graphics+cards")
        end_loop = True
        break

    response_3060 = session.get("https://www.newegg.ca/p/pl?d=RTX+3060+ti+graphics+cards")

    soup_3060 = BeautifulSoup(response_3060.content, 'html.parser')

    if parse_availability(soup_3060, "3060", 1500.00, count):
        notify("3060 Ti", "https://www.newegg.ca/p/pl?d=RTX+3060+ti+graphics+cards")
        end_loop = True
        break

    time.sleep(20)
